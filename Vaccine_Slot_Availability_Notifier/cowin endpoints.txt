# Get all states
https://cdn-api.co-vin.in/api/v2/admin/location/states

# Get all districts within state
https://cdn-api.co-vin.in/api/v2/admin/location/districts/<state_id>

# Get available centres (on a specific date in a given district)
https://cdn-api.co-vin.in/api/v2/appointment/sessions/public/findByDistrict?district_id=<id>&date=<date>

# Get available centres (for 7 days from a specific date in a given district.)
https://cdn-api.co-vin.in/api/v2/appointment/sessions/public/calendarByDistrict?district_id=<id>&date=<date>
